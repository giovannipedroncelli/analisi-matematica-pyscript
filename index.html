import sympy as sp import ziamath as zm import matplotlib.pyplot as plt import numpy as np x = sp.symbols('x') out = Element("out") definito = Element("def") limite = Element("limite") scelte = Element("scelte") scrivi = Element("scrivi") disegno = Element("disegno") aperto=False def istr(): global aperto if aperto: Element("istruzioni").add_class("hidden") aperto=False else: Element("istruzioni").remove_class("hidden") aperto=True def disegna(f): lam_x = sp.lambdify(x, f, modules=['numpy']) assex = np.linspace(-10,10) assey = lam_x(assex) fig,ax = plt.subplots() plt.plot(assex,assey) plt.grid(True) plt.show() disegno.write(fig) def reset(): out.clear() definito.add_class("hidden") limite.add_class("hidden") def conferma_fun(): reset() global f try: f = sp.sympify(Element("funzione").value) scelte.remove_class("hidden") out.remove_class("errore") scrivi.write(zm.Text("Funzione: $y={}$".format(sp.latex(f)))) disegna(f) except: scrivi.clear() disegno.clear() scelte.add_class("hidden") out.write("Funzione in formato errato, riprova.") out.add_class("errore") def conferma_scelta(): reset() selezionato = Element("opzioni").value if selezionato=="DER": der = sp.diff(f, x) out.write(zm.Text("La derivata della funzione è ${}$".format(sp.latex(der)))) elif selezionato=="INTI": inti = sp.integrate(f, x) out.write(zm.Text("L'integrale indefinito della funzione è ${}$".format(sp.latex(inti)))) elif selezionato=="INTD": definito.remove_class("hidden") elif selezionato=="LIM": limite.remove_class("hidden") def conferma_int(): reset() li = Element("inf").element.value ls = Element("sup").element.value intd = sp.integrate(f, (x, li, ls)) out.write(zm.Latex(sp.latex(intd))) def conferma_lim(): reset() t = Element("tende").element.value lim = sp.limit(f,x,t) out.write(zm.Latex(sp.latex(lim)))
